import re

def convert_to_mb(size_str):
    """Converts file size strings like '144mb', '132kb' to numeric MB values."""
    size_str = str(size_str).lower().strip()  # Clean up and normalize case

    
    match = re.match(r'(\d+\.?\d*)\s*(kb|mb)', size_str)
    
    if match:
        size_value = float(match.group(1))
        unit = match.group(2)

        if unit == 'kb':
            # Convert KB to MB
            return size_value / 1024
        elif unit == 'mb':
            return size_value
    return np.nan  # Return NaN for invalid values
